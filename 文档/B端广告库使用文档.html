  • 客户端 直接依赖 adlib即可。
    （adlib内部的关系： adlib —> lib2 —> corelib）

  • 继承AdLibApplication，并在AndroidManifest.xml的Application标签配置。
    或者在自己Application的onCreate()里面添加一句代码:
    Utils.appContext = this;

  • 在需要获取广告的地方创建AdController实例：AdController mAdController = new AdController(<context>);

  • 广告有四种类型 

public static final String TYPE_HOME = "1001"; //'室内机 - 首页'
public static final String TYPE_VIDEO = "1002";  //室内机 - 视频通话
public static final String TYPE_SLEEP = "1003"; //室内机 - 休眠
public static final String TYPE_WAKEUP = "1004"; //室内机 - 唤醒

  • 通过.executeAdRule()获取广告。
    参数1：广告类型
    参数2：familyId（家庭ID）
    参数3：userId (用户ID)
    参数4：ticket
    参数5：接收广告内容的回调接口

  • 获取首页 或 视频通话 广告（仅传递的广告类型不一样） 

mAdController.startPolling(new AdController.IAlarm() {
            @Override
            public void onAlarmReceive() {
                mAdController.executeAdRule(AdController.TYPE_HOME, "442", "1", "", new AdController.IExecuteRule() {
                    @Override
                    public void onAdReceive(AdEntity.AdsEntity adsEntity, File file) {
                        JLog.d("有新的广告来啦. 本地广告路径:" + file + ", 详细信息: " + adsEntity);
                    }

                    @Override
                    public void onAdPlayFinished() {
                        JLog.d("广告播放完成  (有可能是上一次的广告)");
                    }

                    @Override
                    public void onAdDownloadError(AdEntity.AdsEntity adsEntity) {
                        JLog.d("某个广告下载失败: " + adsEntity);
                    }

                    @Override
                    public void onAdExecuteFailed(String reason) {
                        JLog.d("获取广告失败: " + reason);
                    }

                    @Override
                    public void onAdNoExist(String adType, String hour) {
                        JLog.d("没有广告: 广告类型=" + adType + ", 当前小时=" + hour);
                    }
                });
            }
        });

停止获取广告: 

mAdController.stopPolling();
mAdController.cancel(AdController.TYPE_HOME)
  • 获取 唤醒广告 （仅仅在屏幕熄灭情况下才会触发）
    .startWakeUpTimer()会自动唤醒屏幕，并通过接口通知客户端，客户端在回调里面获取对应的广告数据。
mAdController.startWakeUpTimer(new AdController.IWakeUp() {
    @Override
    public void onWakeUpAdExecute() {
        mAdController.executeAdRule(AdController.TYPE_WAKEUP, "442", "1", "", new AdController.IExecuteRule() {
            @Override
            public void onAdReceive(AdEntity.AdsEntity adsEntity, File file) {
                mAdController.wakeUpScreen();
                JLog.d("有新的广告来啦. 本地广告路径:" + file + ", 详细信息: " + adsEntity);
            }

            @Override
            public void onAdPlayFinished() {
                mAdController.turnOffScreen(); //继续保持休眠
                JLog.d("广告播放完成  (有可能是上一次的广告)");
            }

            @Override
            public void onAdDownloadError(AdEntity.AdsEntity adsEntity) {
                JLog.d("某个广告下载失败: " + adsEntity);
            }

            @Override
            public void onAdExecuteFailed(String reason) {
                JLog.d("获取广告失败: " + reason);
            }

            @Override
            public void onAdNoExist(String adType, String hour) {
                JLog.d("没有广告: 广告类型=" + adType + ", 当前小时=" + hour);
            }
        });
    }
});

停止获取广告并停止唤醒屏幕： 

mAdController.cancel(AdController.TYPE_WAKEUP);
mAdController.stopWakeUpTimer();
  • 获取 休眠广告
    .receiveScreenOff() 会接收屏幕休眠的广播并通过回调通知客户端，客户端在回调里面获取广告数据。
    记得手动在onAdReceive()里面调用.wakeUpScreen()唤醒屏幕，在onAdPlayFinished()里面监听广告播放完毕并调用.turnOffScreen()继续休眠屏幕。
mAdController.receiveScreenOff(new AdController.IScreenOff() {
    @Override
    public void onSleepAdExecute() {
        mAdController.executeAdRule(AdController.TYPE_SLEEP, "442", "1", "", new AdController.IExecuteRule() {
            @Override
            public void onAdReceive(AdEntity.AdsEntity adsEntity, File file) {

                mAdController.wakeUpScreen();  //唤醒屏幕

                JLog.d("有新的广告来啦. 本地广告路径:" + file + ", 详细信息: " + adsEntity);
            }

            @Override
            public void onAdPlayFinished() {
            
                mAdController.turnOffScreen(); //继续保持休眠
                
                JLog.d("广告播放完成  (有可能是上一次播放的广告执行完成)");
            }

            @Override
            public void onAdDownloadError(AdEntity.AdsEntity adsEntity) {
                JLog.d("某个广告下载失败: " + adsEntity);
            }

            @Override
            public void onAdExecuteFailed(String reason) {
                JLog.d("获取广告失败: " + reason);
            }

            @Override
            public void onAdNoExist(String adType, String hour) {
                JLog.d("没有广告: 广告类型=" + adType + ", 当前小时=" + hour);
            }
        });
    }
});

停止获取广告并停止接收屏幕休眠的广播：

mAdController.cancel(AdController.TYPE_SLEEP);
mAdController.stopReceiveScreenOff();
